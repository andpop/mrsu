SELECT drama.title,drama.year,count(drama.title) as number_rating FROM ratings INNER JOIN (SELECT title,id,year FROM movies WHERE geners LIKE '%Drama%' AND year > 2005) drama ON movie_id == drama.id INNER JOIN users ON ratings.user_id = users.id WHERE gender == 'female' GROUP BY drama.title ORDER BY drama.year,drama.title;
